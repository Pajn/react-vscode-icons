# Usage

```jsx
import React from 'react';
import Add from 'react-vscode-icons/es/Add';

export default () => {
  return <Add />
}
``` 

## Custom style

```jsx
import React from 'react';
import Add from 'react-vscode-icons/es/Add';

export default () => {
  return <Add style={{fontSize: 24, color: '#fa541c'}}/>
}
```
